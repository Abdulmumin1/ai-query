Learn to build a basic stateful agent that maintains state across interactions.

## What You'll Build

A simple counter agent that remembers its count even after restarts.

## Prerequisites

```bash
pip install ai-query
```

## Step 1: Create a Basic Agent

The simplest stateful agent uses `MemoryStorage`:

```python
import asyncio
from ai_query.agents import Agent, MemoryStorage

class CounterAgent(Agent):
    def __init__(self, agent_id: str):
        super().__init__(
            agent_id,
            storage=MemoryStorage(),
            initial_state={"count": 0}
        )

    async def increment(self):
        await self.update_state(count=self.state["count"] + 1)
        return self.state["count"]

    async def decrement(self):
        await self.update_state(count=self.state["count"] - 1)
        return self.state["count"]

async def main():
    agent = CounterAgent("counter-1")
    async with agent:
        print(f"Initial count: {agent.state['count']}")

        await agent.increment()
        print(f"After increment: {agent.state['count']}")

        await agent.increment()
        await agent.increment()
        print(f"After 2 more increments: {agent.state['count']}")

asyncio.run(main())
```

**Output:**
```
Initial count: 0
After increment: 1
After 2 more increments: 3
```

## Step 2: Add Persistence with SQLite

Switch to `SQLiteStorage` for data that survives restarts:

```python
import asyncio
from ai_query.agents import Agent, SQLiteStorage

class PersistentCounter(Agent):
    def __init__(self, agent_id: str):
        super().__init__(
            agent_id,
            storage=SQLiteStorage("./counters.db"),
            initial_state={"count": 0}
        )

    async def increment(self):
        await self.update_state(count=self.state["count"] + 1)
        return self.state["count"]

async def main():
    # First run
    agent = PersistentCounter("my-counter")
    async with agent:
        print(f"Count: {agent.state['count']}")
        await agent.increment()
        await agent.increment()
        print(f"Count: {agent.state['count']}")

    # Simulate restart - data persists!
    agent2 = PersistentCounter("my-counter")
    async with agent2:
        print(f"After restart: {agent2.state['count']}")

asyncio.run(main())
```

**Output:**
```
Count: 0
Count: 2
After restart: 2
```

## Step 3: Add Lifecycle Hooks

React to agent events:

```python
import asyncio
from ai_query.agents import Agent, SQLiteStorage

class LoggingCounter(Agent):
    def __init__(self, agent_id: str):
        super().__init__(
            agent_id,
            storage=SQLiteStorage("./counters.db"),
            initial_state={"count": 0, "history": []}
        )

    async def on_start(self):
        print(f"Agent started with count: {self.state['count']}")

    async def increment(self):
        new_history = self.state["history"] + [
            {"action": "increment", "value": self.state["count"] + 1}
        ]
        await self.set_state({
            "count": self.state["count"] + 1,
            "history": new_history[-10:]  # Keep last 10
        })

async def main():
    agent = LoggingCounter("logged-counter")
    async with agent:
        await agent.increment()
        await agent.increment()
        print(f"History: {agent.state['history']}")

asyncio.run(main())
```

**Output:**
```
Agent started with count: 0
History: [{'action': 'increment', 'value': 1}, {'action': 'increment', 'value': 2}]
```

## Step 4: Multiple Agent Instances

Each agent ID is independent:

```python
import asyncio
from ai_query.agents import Agent, SQLiteStorage

class UserStats(Agent):
    def __init__(self, agent_id: str):
        super().__init__(
            agent_id,
            storage=SQLiteStorage("./users.db"),
            initial_state={"visits": 0, "name": "Unknown"}
        )

    async def visit(self, name: str = None):
        await self.set_state({
            "visits": self.state["visits"] + 1,
            "name": name or self.state["name"]
        })

async def main():
    # Different users = different states
    alice = UserStats("user-alice")
    async with alice:
        await alice.visit("Alice")
        await alice.visit()
        print(f"Alice: {alice.state}")

    bob = UserStats("user-bob")
    async with bob:
        await bob.visit("Bob")
        print(f"Bob: {bob.state}")

    # Reload Alice - her state is preserved
    alice2 = UserStats("user-alice")
    async with alice2:
        print(f"Alice again: {alice2.state}")

asyncio.run(main())
```

**Output:**
```
Alice: {'visits': 2, 'name': 'Alice'}
Bob: {'visits': 1, 'name': 'Bob'}
Alice again: {'visits': 2, 'name': 'Alice'}
```
